function doPost(e) { try { var data = JSON.parse(e.postData.contents); // 1. จัดการอัปโหลดไฟล์สลิปไปเก็บไว้ที่ Google Drive var slipUrl = "ไม่มีสลิป"; if (data.slipData && data.slipName) { var contentType = data.slipType; var slipBlob = Utilities.newBlob(Utilities.base64Decode(data.slipData), contentType, data.slipName); // นำไปเซฟในโฟลเดอร์หลักของมึง (หรืออยากสร้างโฟลเดอร์แยกก็สามารถทำได้) var file = DriveApp.createFile(slipBlob); // เปิดสิทธิ์ลิงก์ให้มึงกดคลิกดูจาก Google Sheet ได้สะดวก file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); slipUrl = file.getUrl(); } var deposit = parseFloat(data.deposit) || 0; var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var dateTimeStr = data.date + " " + data.time; // 2. บันทึกลงตารางตามคอลัมน์ A - F (และเพิ่ม G เป็นลิงก์ดูสลิป) sheet.appendRow([ data.name, // A: ชื่อ-นามสกุล data.phone, // B: เบอร์โทรศัพท์ data.service, // C: รายละเอียดงาน (รอบนี้คือประเภทงาน เช่น งานปริญญา งานแต่ง) dateTimeStr, // D: วันที่และเวลา deposit, // E: มัดจำ "", // F: คงเหลือ (ปล่อยว่างไว้ให้มึงกรอกราคาเต็มหักลบเองในอนาคตได้เลย) slipUrl // G: ลิงก์รูปภาพสลิปหลักฐาน ]); // 3. ส่งข้อมูลไปสร้างคิวใน Google Calendar var calendars = CalendarApp.getCalendarsByName("Kampon Graphics"); var calendar = calendars.length > 0 ? calendars[0] : CalendarApp.getDefaultCalendar(); var startDateTime = new Date(data.date + "T" + data.time + ":00"); var endDateTime = new Date(startDateTime.getTime() + (1 * 60 * 60 * 1000)); var eventTitle = "📌 [" + data.service + "] คุณ " + data.name; var eventDescription = "📞 เบอร์: " + data.phone + "\n📝 รายละเอียด: " + data.details + "\n💰 มัดจำ: " + deposit + " บาท\n🧾 ลิงก์สลิป: " + slipUrl; calendar.createEvent(eventTitle, startDateTime, endDateTime, { description: eventDescription }); // ตั้งค่าส่งกลับให้หน้าบ้านรู้ว่าทำสำเร็จ var result = JSON.stringify({"status": "success"}); return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); } catch(error) { var result = JSON.stringify({"status": "error", "message": error.toString()}); return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); } }
